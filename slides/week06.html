1.1 Funktion (løsning)


			def ClassifyFruit(shape_round, color, size, sweet):
			    """ (bool, str, str, str, bool) -> str
			        Predicts what kind of fruit the input is
			        through a number of rules
			    """
			    
			    # Set default to unkown
			    fruit = "Unknown"

			    # Green fruits
			    if color == "green" and size == "large":
			        if size == "large":
			            fruit = "watermelon"
			        elif size == "medium":
			            fruit = "apple"
			        else:
			            fruit = "grape"

			    # Yellow fruits
			    elif color == "yellow":
			        if shape_round:
			            if size == "small":
			                fruit = "lemon"
			            else:
			                fruit = "grape fruit"
			        else:
			            fruit = "banana"

			    # Red fruits
			    elif color == "red":
			        if size == "small":
			            if sweet:
			                fruit = "cherry"
			            else:
			                fruit = "grape"
			        else:
			            fruit = "cherry"

			    return fruit

			print(predicted)

			sentences = ["Her er den første sætning", "Dette er den anden sætning", "En fin liste tre sætninger"]
			w = open("sentences.txt", "w")

			for i, sentence in enumerate(sentences):
			    print(i, sentence)
			    w.write(" ".join([str(i), sentence])+"\n")
			    
			    for j, word in enumerate(sentence.split()):
			        print(j, word)
			        w.write(" ".join([str(j), word])+"\n")
			   
			    print()
			    w.write("\n")

			w.close()
